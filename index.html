const H = require('highland')
const R = require('ramda')
const turf = require('@turf/turf')

H(fs.createReadStream('mapwarper.objects.ndjson'))
  .split().compact().map(JSON.parse)
  .filter(onlyNewYorkMapsOfCertainScale)
  .map(toGeoJSONFeature)
  .map((feature) => turf.simplify(feature, 0.00005))
  .map((feature) => turf.buffer(feature, 25, 'meters'))
  .group(determineDecade)
  .map(R.toPairs)
  .sequence()
  .map(R.zipObj(['group', 'features']))
  .map((group) => turf.union(group.features))
  .toArray((features) => {
    console.log({type: 'FeatureCollection', features})
  })
